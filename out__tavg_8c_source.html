55  {

58  }
59  /* sum */
60  for (i = 0; i < nx; i++)

62  for (i = 0; i < nc; i++)
63  d->c[i] += c[i];
64  d->t += t;
65  d->pos++;
66  if (d->pos==d->len) {

71  d->c[i] /= d->len;
72  d->t /= d->len;
73  /* pass on to next output */
74  if ((stat = d->next_out->apply(d->next_out, d->t, nx, d->x, nc, d->c)) != SD_CONT)
75  return stat;
